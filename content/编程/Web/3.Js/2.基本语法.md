## 数组

```javascript
/ 数组
let fruits = ["apple", "banana", "orange"];
// 访问数组所有元素
console.log(fruits);

// 访问数组第一个元素
console.log(fruits[0]);

//  在数组末尾添加元素
fruits.push("pear");
console.log(fruits);

// 在数组开头添加元素
fruits.unshift("grape");
console.log(fruits);

// 删除数组末尾元素
fruits.pop();
console.log(fruits);

// 查看变量是否为数组
console.log(Array.isArray(fruits));

// 差看元素索引
console.log(fruits.indexOf("banana"));
```

### 对象

```javascript
// 对象
let person = {
    name: "ty",
    age: 21,
    gender: "male",
    fruits: ["apple", "banana", "orange"],
    address: {
        city: "beijing",
        street: "chaoyang"
    }
}

// 访问对象属性
console.log(person.name);
console.log(person.address.city);
console.log(person.fruits[0]);

// 抽取对象属性
let { gender , address: { city } , fruits: [fruit1, fruit2] } = person;
console.log(gender, city, fruit1, fruit2);
```

### if

```javascript
let x = 10;
if (x > 10) {
    console.log("x > 10");
}
else if (x < 10) {
    console.log("x < 10");
}
else {
    console.log("x = 10");
}
```

###  三元运算符

```javascript
// 三元运算符
let y = 10;
console.log(y > 10 ? "y > 10" : "y <= 10");
// 这里的？是条件，：是else
```

### switch

```javascript
// switch语句
let z = 10;
switch (z) {
    case 10:
        console.log("z = 10");
        break;
    case 20:
        console.log("z = 20");
        break;
    default:
        console.log("z = other");
        break;
}
```

### 循环

```javascript

// 循环语句
// for循环
for (let i = 0; i < 10; i++) {
    console.log(i);
}

// while循环
let j = 0;
while (j < 10) {
    console.log(j);
    j++;
}
```
